                  document.addEventListener('DOMContentLoaded', function() {
                  	const seasonBtns = document.querySelectorAll('.season-btn');
                  	const wrapper = document.querySelector('.season-buttons-wrapper');
                  	const container = document.querySelector('.season-buttons');
                  
                  	// Check for overflow and scroll position
                  	function checkOverflow() {
                  		if (container && wrapper) {
                  			const scrollArrow = document.querySelector('.scroll-arrow');
                  			if (container.scrollWidth > container.clientWidth) {
                  				// Check if scrolled to end (last button visible)
                  				const isScrolledToEnd = container.scrollLeft + container.clientWidth >= container.scrollWidth - 1;
                  
                  				if (isScrolledToEnd) {
                  					// Hide both arrow and fade when at end
                  					wrapper.classList.remove('has-overflow');
                  					if (scrollArrow) {
                  						scrollArrow.style.opacity = '0';
                  						scrollArrow.style.pointerEvents = 'none';
                  					}
                  				} else {
                  					// Show both arrow and fade when not at end
                  					wrapper.classList.add('has-overflow');
                  					if (scrollArrow) {
                  						scrollArrow.style.opacity = '1';
                  						scrollArrow.style.pointerEvents = 'auto';
                  					}
                  				}
                  			} else {
                  				wrapper.classList.remove('has-overflow');
                  				if (scrollArrow) {
                  					scrollArrow.style.opacity = '0';
                  					scrollArrow.style.pointerEvents = 'none';
                  				}
                  			}
                  		}
                  	}
                  
                  	// Initial check
                  	checkOverflow();
                  
                  	// Check on window resize
                  	window.addEventListener('resize', checkOverflow);
                  
                  	// Check on scroll
                  	container.addEventListener('scroll', checkOverflow);
                  
                  	// Scroll arrow functionality
                  	const scrollArrow = document.querySelector('.scroll-arrow');
                  	if (scrollArrow) {
                  		scrollArrow.addEventListener('click', function() {
                  			const scrollAmount = container.clientWidth * 0.8; // Scroll 80% of visible width
                  			container.scrollBy({
                  				left: scrollAmount,
                  				behavior: 'smooth'
                  			});
                  		});
                  	}
                  
                  	// Function to switch season
                  	function switchToSeason(btn) {
                  		// Remove active from all buttons
                  		seasonBtns.forEach(b => b.classList.remove('active'));
                  
                  		// Add active to clicked button
                  		btn.classList.add('active');
                  
                  		// Update hidden season number display
                  		const seasonNum = btn.getAttribute('data-season');
                  		const nSElement = document.querySelector('.n_s');
                  		if (nSElement) {
                  			nSElement.textContent = seasonNum;
                  		}
                  
                  		// Store in localStorage
                  		const postId = btn.getAttribute('data-post');
                  		try {
                  			localStorage.setItem('lastSeasonSelected_' + postId, seasonNum);
                  		} catch(e) {
                  			// localStorage not available
                  		}
                  
                  		// Trigger the original AJAX season switching
                  		const episodeContainer = document.querySelector('#episode_by_temp');
                  		if (episodeContainer) {
                  			episodeContainer.innerHTML = '<div style="text-align: center; padding: 20px;">Loading episodes...</div>';
                  		}
                  
                  		// Make AJAX call using GET for Cloudflare caching
                  		jQuery.ajax({
                  			url: torofilm_Public.url + '?action=action_select_season&season=' + seasonNum + '&post=' + postId,
                  			method: 'GET',
                  			cache: true,
                  			beforeSend: function() {
                  				const html = '<li style="flex:100%; max-width: 100%;"><div style="width:100%; display:flex; justify-content:center"><i id="sl-home" class="fas fa-spinner"></i></div></li>';
                  				if (episodeContainer) {
                  					episodeContainer.innerHTML = html;
                  				}
                  			},
                  			success: function(data) {
                  				if (episodeContainer) {
                  					episodeContainer.innerHTML = data;
                  				}
                  			}
                  		});
                  	}
                  
                  	// Add click handlers to season buttons
                  	seasonBtns.forEach(btn => {
                  		btn.addEventListener('click', function() {
                  			switchToSeason(this);
                  		});
                  	});
                  
                  	// Check if user has a previously selected season for this series
                  	if (seasonBtns.length > 0) {
                  		const firstBtn = seasonBtns[0];
                  		const postId = firstBtn.getAttribute('data-post');
                  		let lastSelectedSeason = null;
                  
                  		try {
                  			lastSelectedSeason = localStorage.getItem('lastSeasonSelected_' + postId);
                  		} catch(e) {
                  			// localStorage not available
                  		}
                  
                  		// If user has previously selected a season, switch to it
                  		if (lastSelectedSeason) {
                  			const targetBtn = Array.from(seasonBtns).find(btn => btn.getAttribute('data-season') === lastSelectedSeason);
                  			if (targetBtn) {
                  				// Remove active from current default season
                  				seasonBtns.forEach(b => b.classList.remove('active'));
                  				// Switch to last selected season
                  				switchToSeason(targetBtn);
                  			}
                  		}
                  	}
                  });
